278 void cr_idle(void)
279 {
280  // This needs to be static because the function initially returns normally.
281  static cr_id_t temp_id;
282 
283  CR_THREAD_INIT();
284 
285  assert((this_id__ == CR_IDLE_THREAD_ID) &&
286  "cr_idle: this_id__ != CR_IDLE_THREAD_ID!\n");
287 
288  // This will be the entry point when longjump is called with cr_idle's context.
289  // No need to perform another setjmp from within the loop; not much happening.
290  for (;;) {
291  // Spin until a non-idle thread is activated.
292  if (cr_g_activate_id == CR_IDLE_THREAD_ID)
293  continue;
294 
295  assert((cr_g_activate_id != CR_INVALID_ID) &&
296  "cr_idle: cr_g_activate_id == CR_INVALID_ID!\n");
297  assert(((uint32_t) cr_g_activate_id <= cr_g_context_cnt) &&
298  "cr_idle: cr_g_activate_id out of bounds!\n");
299  assert((cr_g_activate_id != this_id__) &&
300  "cr_idle: recursive coroutine call!\n");
301 
302  temp_id = cr_g_activate_id;
303  cr_g_activate_id = CR_IDLE_THREAD_ID;
304  cr_g_previous_cr_id = CR_IDLE_THREAD_ID;
305 
306  if (!setjmp(cr_g_context[CR_IDLE_THREAD_ID].env))
307  longjmp(cr_g_context[temp_id].env, SETJMP_DFLT_RET_VAL);
308  else
309  /* explicit block for the longjmp */ ;
310  }
311 }

327 cr_id_t cr_register_thread(void(*pFunc)(void))
328 {
329  // Increase the coroutine count.
330  cr_g_thread_cnt += 1;
331 
332  if (cr_g_thread_cnt == CR_IDLE_THREAD_ID)
333  if (pFunc != cr_idle)
334  perror("cr_idle error: cr_g_thread_cnt != CR_IDLE_THREAD_ID. \
335  Note, coroutines shouldn't be registered prior to calling cr_init.");
336 
337  assert(pFunc && "cr_register_thread: pFunc is invalid!\n");
338 
339  // Account fot the cr_idle coroutine (one reserved index in the array) in the assert check.
340  assert((cr_g_thread_cnt < cr_g_context_cnt) &&
341  "cr_register_thread: [ cr_g_thread_cnt < cr_g_context_cnt ] failed\n");
342 
343  // Assign the coroutine's ID and function pointer to its context structure
344  cr_g_context[cr_g_thread_cnt].id = cr_g_thread_cnt;
345  cr_g_context[cr_g_thread_cnt].pFunc = pFunc;
346 
347  // cr_g_current_cr_id is used by the CR_THREAD_INIT macro to identify
348  // the coroutine.
349  cr_g_current_cr_id = cr_g_thread_cnt;
350 
351  if (!setjmp(cr_g_reg_func_env))
352  pFunc(); // this function won't return normally
353  else
354  /* explicit block for the longjmp */ ;
355 
356  // The function is now a coroutine.
357  // This is where we return from CR_THREAD_INIT via the longjump.
358 
359  // A sentinal. At init the coroutine returned via a longjump so the return
360  // point saved in the function's prolog is valid and will return here
361  // should the coroutine actually return normally via the epilog code.
362  if (cr_g_sys_started == CR_SYSTEM_STARTED) {
363  // What to do?
364  // - nothing and return normally
365  // - yield to whatever is in cr_g_previous_cr_id
366  // - yield to cr_idle
367  // - assert
368  // - exit
369 
370  // Note, cr_g_current_cr_id holds the ID of the exiting coroutine.
371  // Note, cr_g_previous_cr_id holds the ID of the previous coroutine.
372 
373  // Note, it's not possible to remove the exiting coroutine from the
374  // context array because the remaining coroutine ID's cannot be changed,
375  // a coroutine's ID is the index into the context array.
376 
377  // Look at the cr_g_current_cr_id variable in a debugger to see the ID
378  // of the exiting coroutine.
379  assert(CR_ERROR_CR_EXITING &&
380  "cr_register_thread: coruotine exiting!\n");
381 
382  // Pass EXIT_SUCCESS; we want any user registered clean-up functions to be called.
383  exit(EXIT_SUCCESS);
